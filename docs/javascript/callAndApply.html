# JavaScript之Call、Apply和bind

# call的实现

先来看call的特点

  • call的第一个参数是调用的函数中this指向的对象，如果没有传入（如null或undefined）则指向window
  • call的第一个参数如果是基本数据类型，则指向该数据类型的类（如1为Number类）
  • call后面的参数以逗号分割，是传入的调用函数的参数

实现

Function.prototype.$call = function(){
    var firstArg = arguments[0];
    var obj;
    if(firstArg === null || firstArg === undefined){
         obj =  window;
    }else{
         obj =  Object(firstArg);
    }
    var args = [];
    for(var i=1,len=arguments.length;i<len;i++){
      args.push("arguments["+i+"]")
    }

    obj.$fun = this;
    eval("obj.$fun(" + args +")")
    
    delete obj.$fun;
}


// 测试使用
var obj = {
    name:"121212"
};
function test(a,b){
    console.log(this);
    console.log(a);
    console.log(b);
}

test.$call(obj,"哈哈","324")

这里的需要注意的知识点：

  • 字符串与数组相加，数组会先调用toString()方法
// 调用toString()方法
var ary = [1,2,3];
var str = "fun("+ ary + ")";  //返回 "fun(1,2,3)"

// 调用toString()方法 但是这里有个问题
// 转化后是这样"fun(哈哈,hh,3)"
// 括号中的`哈哈`没有引号，`hh`没有引号, 3也被转化成没有引号的3
// 这时被eval解析时，就会报错，我们想要的应该是"fun('哈哈','hh','3')"这样
var arys = ["哈哈","hh","3"];
var strs = "fun("+ ary + ")";  // "fun(哈哈,hh,3)"

针对上面情况我们做了这个操作args.push("arguments["+i+"]"), 当eval执行时会自动将字符串形式的"arguments[1]" 解析成对应的值，

# apply的实现

我们在代码中并没有做一些参数判断，例如第二个参数必须是个数组

Function.prototype.$apply = function(){
    var firstArg = arguments[0];
    var obj;
    if(firstArg === null || firstArg === undefined){
         obj =  window;
    }else{
         obj =  Object(firstArg);
    }
   
    var argAry =  arguments[1] || [];
    var args = [];
    for(var i=0,len=argAry.length;i<len;i++){
        args.push("argAry["+i+"]")
    }

    obj.$fun = this;
    eval("obj.$fun(" + args +")")

    delete obj.$fun;
}


var obj = {
    name:"121212"
};

function test(a,b){
    console.log(this);
    console.log(a);
    console.log(b);
}


test.$apply(obj,["12","324"])

# bind的实现

Function.prototype.$bind = function(){
    var self = this;

    var args = Array.prototype.slice.call(arguments); 
    var obj = args[0];

    if(obj === null || obj === undefined){
        obj =  window;
    }else{
        obj =  Object(obj);
    }
    args.splice(0,1)

    return function(s){ // 使用柯里化函数
        args.push(s)
        console.log(args)
        self.apply(obj,args);
    }
}

var obj = {
    name:"121212"
};

function test(a,b){
    console.log(this);
    console.log(a);
    console.log(b);
}

var fun = test.$bind(obj,1)

fun(34)
